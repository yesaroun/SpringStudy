/**
* Member 테이블 : 엔티티
*/
public class Member {
    /** id 칼럼 대응 */
    private Integer id;

    /** name 칼럼 대응*/
    private String name;

    public Integer getId() {
        return id;
    }

    public void setId(Integer id) {
        this.id = id;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }
}
